def is_game_over(nb_sticks_used):
    if nb_sticks_used == 5:
        return True
    return False

def evaluate_board (is_ai_turn):
    if (is_ai_turn == True):
        return 1
    return (-1)

def simulate_future(nb_sticks_used, is_ai_turn):
    if is_game_over(nb_sticks_used):
        return evaluate_board(is_ai_turn)
    
    if is_ai_turn == True:
        best_score = -100
    else:
        best_score = 100

    for choice in range(1, 3):
        if nb_sticks_used + choice <=5 :
            simulated_sticks = nb_sticks_used + choice
            score = simulate_future(simulated_sticks, not is_ai_turn)
            if is_ai_turn == True :

                if score > best_score:
                    best_score = score
                    final_choice = choice
            else : 

                if score < best_score:
                    best_score = score 
                    final_choice = choice
    return best_score
